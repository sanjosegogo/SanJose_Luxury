/** * 這是部署在 Google Apps Script 的程式碼 * 功能：接收請求並回傳 JSON 格式的試算表資料 */ function doGet(e) { const action = e.parameter.action; const ss = SpreadsheetApp.getActiveSpreadsheet(); // 1. 處理跑馬燈資料 if (action === 'marquee') { const sheet = ss.getSheetByName('跑馬燈'); // 請確保分頁名稱正確 const data = sheet.getDataRange().getValues(); const messages = data.flat().filter(m => m && m.toString().trim() !== ""); return JSON_RESPONSE(messages); } // 2. 處理商品主資料 const sheet = ss.getSheetByName('商品資料'); // 請確保分頁名稱正確 const rows = sheet.getDataRange().getValues(); // 移除標題列 rows.shift(); return JSON_RESPONSE(rows); } // 統一 JSON 回傳格式並解決跨網域 (CORS) 問題 function JSON_RESPONSE(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
